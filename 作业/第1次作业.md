   &{\begin{vmatrix}
   2&0&1\\
   1&-4&-1\\
   -1&8&3
   \end{vmatrix}}
   =\frac12
   {\begin{vmatrix}
   2&0&1\\
   2&-8&-2\\
   -1&8&3
   \end{vmatrix}}
   =\frac12
   {\begin{vmatrix}
   2&0&1\\
   0&-8&-3\\
   -1&8&3
   \end{vmatrix}}\\[1em]
   
   =\frac14
   &{\begin{vmatrix}
   2&0&1\\
   0&-8&-3\\
   -2&16&6
   \end{vmatrix}}
   =\frac14
   {\begin{vmatrix}
   2&0&1\\
   0&-8&-3\\
   0&16&7
   \end{vmatrix}}
   =\frac18
   {\begin{vmatrix}
   2&0&1\\
   0&-16&-6\\
   0&16&7
   \end{vmatrix}}\\[1em]
   
   =\frac18
   &{\begin{vmatrix}
   2&0&1\\
   0&-16&-6\\
   0&0&1
   \end{vmatrix}}
   =\frac18\times2\times(-16)\times1\\[1em]
   
   =-&4
   \end{align}
   $$

   a&b&c\\

   \end{vmatrix}
   }
   =\frac1{a^2}
   {
   \begin{vmatrix}
   a&b&c\\
   0&ac-b^2&a^2-bc\\
   0&a^2-bc&ab-c^2
   \end{vmatrix}
   }\\[1em]
   
   =\frac1{a^3}&
   {
   \begin{vmatrix}
   a&b&c\\
   0&ac-b^2&a^2-bc\\

   \end{vmatrix}
   }
   =\frac1{a^3}

   \begin{vmatrix}
   a&b&c\\
   0&ac-b^2&a^2-bc\\

   \end{vmatrix}
   }\\[1em]
   
   =a^3&+3abc-b^3
   \end{align}

   \end{vmatrix}}\\[1em]
